SET Output_Status TO 200
BLOCK 
ON BLOCK ERROR all

END
    System.GetEnvironmentVariable.GetEnvironmentVariable Name: $'''USERNAME''' Value=> UserName
    IF (System.IfProcess.ProcessWithNameIsNotRunning ProcessName: $'''EXCEL''' ProcessUser: UserName) THEN
    ON ERROR

    END
        SET Output_Message TO $'''Excel is not running. Will not attempt to close it.'''
        GOTO 'Exit'
    END
    IF IsNotEmpty(Input_ExcelInstance) THEN
        IF Input_ExcelInstance['Handle'] <> 0 THEN
            Excel.CloseExcel.Close Instance: Input_ExcelInstance
            WAIT (System.WaitForProcess.ProcessWithNameToStop ProcessName: $'''EXCEL''' ProcessUser: UserName) FOR 10 
            ON ERROR TimeoutError

            END
            IF (System.IfProcess.ProcessWithNameIsNotRunning ProcessName: $'''EXCEL''' ProcessUser: UserName) THEN
            ON ERROR

            END
                SET Output_Message TO $'''Successfully closed Excel instance.'''
                GOTO 'Exit'
            END
        END
    END
    LOOP FROM 1 TO 10 STEP 1
        IF (System.IfProcess.ProcessWithNameIsRunning ProcessName: $'''EXCEL''' ProcessUser: UserName) THEN
        ON ERROR

        END
            Scripting.RunPowershellScript.RunPowershellScript Script: $'''$excel = [Runtime.Interopservices.Marshal]::GetActiveObject(\'Excel.Application\')
$excel.Visible = $false
$excel.DisplayAlerts = $false

$excel.Quit()
[System.Runtime.Interopservices.Marshal]::ReleaseComObject($excel)
Remove-Variable -Name Excel 
[GC]::Collect() 
[GC]::WaitForPendingFinalizers()
if ($excel.Workbooks.Count -eq 0)
{
    TASKKILL.EXE /FI \'USERNAME eq $Env:UserName\' /IM EXCEL.EXE
}''' ScriptOutput=> PowershellOutput
            WAIT (System.WaitForProcess.ProcessWithNameToStop ProcessName: $'''EXCEL''' ProcessUser: UserName) FOR 5 
            ON ERROR

            END
        ELSE
            EXIT LOOP
        END
    END
    LOOP FROM 1 TO 10 STEP 1
        IF (System.IfProcess.ProcessWithNameIsRunning ProcessName: $'''EXCEL''' ProcessUser: UserName) THEN
        ON ERROR

        END
            System.TerminateProcess.TerminateProcessByNameWithUserName ProcessUser: UserName ProcessName: $'''EXCEL'''
            ON ERROR

            END
            WAIT (System.WaitForProcess.ProcessWithNameToStop ProcessName: $'''EXCEL''' ProcessUser: UserName) FOR 5 
            ON ERROR

            END
        ELSE
            EXIT LOOP
        END
    END
    IF (System.IfProcess.ProcessWithNameIsRunning ProcessName: $'''EXCEL''' ProcessUser: UserName) THEN
    ON ERROR

    END
        SET Output_Status TO 500
        SET ErrorMessage TO $'''Failed to properly close excel'''
        GOTO 'Exit'
    END
    SET Output_Message TO $'''Successfully closed Excel'''
    LABEL 'Exit'
    EXIT Code: 0
END
ERROR => ErrorMessage
SET Output_Status TO 500
